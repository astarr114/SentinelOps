   204 /**
   205  * Robustly extract the first valid JSON object from LLM output.
   206  * Handles: bare JSON, ```json fences, ```  fences, leading/trailing prose.
   207  */
   208 function extractJson(text: string): Record<string, unknown> | null {
   209   if (!text?.trim()) return null;
   210   // Strip markdown code fences (```json ... ``` or ``` ... ```)
   211   const stripped = text.replace(/^```(?:json)?\s*/im, "").replace(/\s*```\s*$/im, "").trim();
   212   // Try the full stripped text first
   213   for (const candidate of [stripped, text]) {
   214     // Find the outermost { ... } (greedy)
   215     const m = /\{[\s\S]*\}/.exec(candidate);
   216     if (!m) continue;
   217     let jsonStr = m[0];
   218     // Attempt direct parse
   219     try { return JSON.parse(jsonStr) as Record<string, unknown>; } catch { /* fall through */ }
   220     // Strip trailing commas before } or ] (common LLM mistake)
   221     jsonStr = jsonStr.replace(/,\s*([}\]])/g, "$1");
   222     try { return JSON.parse(jsonStr) as Record<string, unknown>; } catch { /* fall through */ }
   223   }
   224   return null;
   225 }
